-- Tables create table profiles ( id uuid primary key references auth.users, username text unique not null, display_name text not null, bio text default '', color text default '#5b8fff', avatar_url text, created_at timestamptz default now() ); create table posts ( id uuid primary key default gen_random_uuid(), user_id uuid references profiles not null, username text not null, display_name text not null, color text not null, body text not null default '', image_url text, like_count int default 0, reply_count int default 0, parent_id uuid references posts default null, created_at timestamptz default now() ); create table likes ( user_id uuid references profiles, post_id uuid references posts, primary key (user_id, post_id) ); create table follows ( follower_id uuid references profiles, following_id uuid references profiles, primary key (follower_id, following_id) ); -- RLS alter table profiles enable row level security; alter table posts enable row level security; alter table likes enable row level security; alter table follows enable row level security; create policy "pub read" on profiles for select using (true); create policy "own insert" on profiles for insert with check (auth.uid()=id); create policy "own update" on profiles for update using (auth.uid()=id); create policy "pub read" on posts for select using (true); create policy "auth insert" on posts for insert with check (auth.uid() is not null); create policy "auth update" on posts for update using (auth.uid() is not null); create policy "own delete" on posts for delete using (auth.uid()=user_id); create policy "pub read" on likes for select using (true); create policy "auth like insert" on likes for insert with check (auth.uid()=user_id); create policy "auth like delete" on likes for delete using (auth.uid()=user_id); create policy "pub read" on follows for select using (true); create policy "auth fol insert" on follows for insert with check (auth.uid()=follower_id); create policy "auth fol delete" on follows for delete using (auth.uid()=follower_id); -- Trigger: auto-update like_count (runs as DB owner, bypasses RLS) create or replace function trg_like_count() returns trigger language plpgsql security definer as $$ begin if TG_OP='INSERT' then update posts set like_count = like_count + 1 where id = NEW.post_id; else update posts set like_count = greatest(0, like_count - 1) where id = OLD.post_id; end if; return null; end;$$; create trigger on_like after insert or delete on likes for each row execute function trg_like_count(); -- Trigger: auto-update reply_count create or replace function trg_reply_count() returns trigger language plpgsql security definer as $$ begin if TG_OP='INSERT' and NEW.parent_id is not null then update posts set reply_count = reply_count + 1 where id = NEW.parent_id; elsif TG_OP='DELETE' and OLD.parent_id is not null then update posts set reply_count = greatest(0, reply_count - 1) where id = OLD.parent_id; end if; return null; end;$$; create trigger on_reply after insert or delete on posts for each row execute function trg_reply_count(); -- Public storage bucket for images and GIFs insert into storage.buckets (id, name, public) values ('chirp', 'chirp', true) on conflict (id) do nothing; create policy "pub read chirp" on storage.objects for select using (bucket_id='chirp'); create policy "auth upload chirp" on storage.objects for insert with check (bucket_id='chirp' and auth.uid() is not null); create policy "own delete chirp" on storage.objects for delete using (bucket_id='chirp' and auth.uid()::text = owner);
